48  void BeginSession(const std::string& name, const std::string& filepath = "results.json")
49  {
50  std::lock_guard lock(m_Mutex);
51  if (m_CurrentSession)
52  {
53  // If there is already a current session, then close it before beginning new one.
54  // Subsequent profiling output meant for the original session will end up in the
55  // newly opened session instead. That's better than having badly formatted
56  // profiling output.
57  if (Log::GetCoreLogger()) // Edge case: BeginSession() might be before Log::Init()
58  {
59  GE_CORE_ERROR("Instrumentor::BeginSession('{0}') when session '{1}' already open.", name, m_CurrentSession->Name);
60  }
61  InternalEndSession();
62  }
63  m_OutputStream.open(filepath);
64 
65  if (m_OutputStream.is_open())
66  {
67  m_CurrentSession = new InstrumentationSession({ name });
68  WriteHeader();
69  }
70  else
71  {
72  if (Log::GetCoreLogger()) // Edge case: BeginSession() might be before Log::Init()
73  {
74  GE_CORE_ERROR("Instrumentor could not open results file '{0}'.", filepath);
75  }
76  }
77  }

79  void EndSession()
80  {
81  std::lock_guard lock(m_Mutex);
82  InternalEndSession();
83  }

85  void WriteProfile(const ProfileResult& result)
86  {
87  std::stringstream json;
88 
89  json << std::setprecision(3) << std::fixed;
90  json << ",{";
91  json << "\"cat\":\"function\",";
92  json << "\"dur\":" << (result.ElapsedTime.count()) << ',';
93  json << "\"name\":\"" << result.Name << "\",";
94  json << "\"ph\":\"X\",";
95  json << "\"pid\":0,";
96  json << "\"tid\":" << result.ThreadID << ",";
97  json << "\"ts\":" << result.Start.count();
98  json << "}";
99 
100  std::lock_guard lock(m_Mutex);
101  if (m_CurrentSession)
102  {
103  m_OutputStream << json.str();
104  m_OutputStream.flush();
105  }
106  }

154  class InstrumentationTimer

156  public:
157  InstrumentationTimer(const char* name)
158  : m_Name(name), m_Stopped(false)

160  m_StartTimepoint = std::chrono::steady_clock::now();

169  void Stop()
170  {
171  auto endTimepoint = std::chrono::steady_clock::now();
172  auto highResStart = FloatingPointMicroseconds{ m_StartTimepoint.time_since_epoch() };
173  auto elapsedTime = std::chrono::time_point_cast<std::chrono::microseconds>(endTimepoint).time_since_epoch() - std::chrono::time_point_cast<std::chrono::microseconds>(m_StartTimepoint).time_since_epoch();
174 
175  Instrumentor::Get().WriteProfile({ m_Name, highResStart, elapsedTime, std::this_thread::get_id() });
176 
177  m_Stopped = true;
178  }
179  private:
180  const char* m_Name;
181  std::chrono::time_point<std::chrono::steady_clock> m_StartTimepoint;
182  bool m_Stopped;
183  };

187  template <size_t N>
188  struct ChangeResult
189  {
190  char Data[N];
191  };
192 
193  template <size_t N, size_t K>
194  constexpr auto CleanupOutputString(const char(&expr)[N], const char(&remove)[K])
195  {
196  ChangeResult<N> result = {};
197 
198  size_t srcIndex = 0;
199  size_t dstIndex = 0;
200  while (srcIndex < N)
201  {
202  size_t matchIndex = 0;
203  while (matchIndex < K - 1 && srcIndex + matchIndex < N - 1 && expr[srcIndex + matchIndex] == remove[matchIndex])
204  matchIndex++;
205  if (matchIndex == K - 1)
206  srcIndex += matchIndex;
207  result.Data[dstIndex++] = expr[srcIndex] == '"' ? '\'' : expr[srcIndex];
208  srcIndex++;
209  }
210  return result;
211  }
212  }
213 }
